Frontend: React + TypeScript

// Real-time event streaming with Server-Sent Events
const useRunStore = create<RunState>((set, get) => ({
  events: [],
  async startRun(cfg: CampaignConfig) {
    const res = await fetch("/api/campaigns", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(cfg)
    });
    const { run_id } = await res.json();
    set({ currentRunId: run_id, events: [], isRunning: true });
    get().connectEvents(run_id);
  },
  
  connectEvents(runId: string) {
    const es = new EventSource(`/api/campaigns/${runId}/events`);
    es.onmessage = (e) => {
      const event = JSON.parse(e.data);
      set(state => ({ 
        events: [...state.events, event].slice(-1000)
      }));
    };
  }
}));

The frontend uses Zustand for state management and implements real-time event streaming with Server-Sent Events for live campaign monitoring.
